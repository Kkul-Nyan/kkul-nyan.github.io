### 8.각도기
```swift
import Foundation

func solution(_ angle:Int) -> Int {
    var answer:Int
    if(angle > 0 && angle < 90 ) {answer = 1}
    else if(angle == 90 ) {answer = 2}
    else if(angle > 90 && angle < 180) {answer = 3}
    else {answer = 4}
    return answer
}
```

### 10.짝수의 합
```swift
import Foundation

func solution(_ n:Int) -> Int {
    let num = n/2
    var num1 = 0
    var num2 = 0
    while(num1 < num){
        num1 += 1
        num2 += num1
    }
    var answer = num2 * 2
    return answer
}
```
